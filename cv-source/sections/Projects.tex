\begin{rubric}{Applied Research Tools}

\entry*[2021--Present]%
  \textbf{\href{https://github.com/DVBeckwitt/ra_sim}{2D Powder Simulator}}%
  \begin{itemize}
    \item Simulates detector-space diffraction from 2D-oriented powder films with geometry and model parameters tied to area-detector images.
  \end{itemize}

\entry*[2021--Present]%
  \textbf{\href{https://github.com/DVBeckwitt/2D_Mosaic_Sim}{2D Powder Visualizer}}%
  \begin{itemize}
    \item Visualizes how orientation distributions and mosaic spread reshape two-dimensional diffraction patterns.
  \end{itemize}

\entry*[2021--Present]%
  \textbf{\href{https://github.com/DVBeckwitt/OSC_Reader}{Detector Data Pipeline}}%
  \begin{itemize}
    \item Reads proprietary OSC detector images and converts them into accessible formats for inspection and downstream analysis.
  \end{itemize}

\end{rubric}
