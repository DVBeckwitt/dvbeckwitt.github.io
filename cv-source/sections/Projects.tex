\begin{rubric}{Applied Research Tools}
\entry*[2025--Present]%
  \textbf{\href{https://github.com/DVBeckwitt/2D_Mosaic_Sim}{2D\_Mosaic\_Sim}} – X-ray Diffraction Simulator%
  \begin{itemize}
    \item Visualizes diffraction patterns for materials with specific crystal orientations.
    \item Supports analysis beyond standard software through custom plotting and interpretation tools.
  \end{itemize}

  \entry*[2024--Present]%
  \textbf{\href{https://github.com/DVBeckwitt/ra_sim}{ra\_sim}} – Crystal Analysis Software%
  \begin{itemize}
    \item Simulates and analyzes diffraction data from R-Axis IV++ detectors.
    \item Combines a focused interface with optimization methods tailored to this detector format.
  \end{itemize}

  \entry*[2024]%
  \textbf{\href{https://github.com/DVBeckwitt/OSC_Reader}{OSC\_Reader}} – Detector Data Converter%
  \begin{itemize}
    \item Converts proprietary detector files into accessible formats.
    \item Enables detailed inspection of diffraction images without commercial software.
  \end{itemize}

\end{rubric}
